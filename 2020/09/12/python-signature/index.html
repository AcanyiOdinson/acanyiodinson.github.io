前情提要

公司接口测试需要对接Java，测试需要对接一套Python的签名算法

class SignatureModel:

    def __init__(self):

        self.header = {"Content-type": "application/json"}

    # 全局签名请求格式签名
    def createsignature(self, passParameter):

        # 构造trace_id,去'-'
        traceId = str(uuid.uuid4()).replace('-', '')

        # 当前时间戳!
        timestamp = str(round(time.time() * 1000))

        # 根据客户端的不同选择不同的signKey,secret_key
        signKey = ''
        secretKey = ''

        # 默认参数,每个接口必须请求的参数
        defaultParameter = {'signKey': signKey, 'timestamp': timestamp, 'traceId': traceId}

        # 合并传进来的参数及默认参数
        defaultParameter.update(passParameter)

        # 排序拼接key和value
        sequence = ''
        for key in sorted(defaultParameter.keys()):

            # 如果参数长度大于5000，则截取前5000位
            if len(str(defaultParameter[key])) > 5000:

                # 如果传参为列表,需要对list转换成str类型,再将单引号转换成双引号
                sequence = sequence + key + str(defaultParameter[key][0:5000]).replace("'", '"').replace(' ', '')

            else:

                # 如果传参为列表,需要对list转换成str类型,再将单引号转换成双引号
                sequence = sequence + key + str(defaultParameter[key]).replace("'", '"').replace(' ', '')

        # 拼接密钥
        sequence = sequence + secretKey
        print("签名数据:\n{}".format(sequence))

        # md5加密
        h = hashlib.md5()
        h.update(sequence.encode('utf8'))
        signature = h.hexdigest()[4:34].upper()

        defaultParameter['signature'] = signature

        return defaultParameter
